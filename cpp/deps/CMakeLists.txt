set(LIBRARY_TARGET_NAME ${PROJECT_NAME}_matplotlibcpp)

add_library(${LIBRARY_TARGET_NAME} INTERFACE)
add_library(${PROJECT_NAME}::${LIBRARY_TARGET_NAME} ALIAS ${LIBRARY_TARGET_NAME})
target_include_directories(${LIBRARY_TARGET_NAME} INTERFACE $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>)
target_include_directories(${LIBRARY_TARGET_NAME} INTERFACE $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>)
target_include_directories(${LIBRARY_TARGET_NAME} INTERFACE ${PYTHON_INCLUDE_DIRS})
target_link_libraries(${LIBRARY_TARGET_NAME} INTERFACE ${PYTHON_LIBRARIES})
